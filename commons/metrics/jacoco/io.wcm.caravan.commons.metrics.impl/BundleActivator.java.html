package io.wcm.caravan.commons.metrics.impl;

import java.util.Hashtable;

import org.osgi.framework.BundleContext;
import org.osgi.framework.ServiceRegistration;

import com.codahale.metrics.MetricRegistry;
import com.codahale.metrics.health.HealthCheckRegistry;

/**
 * Registers metric and health check registry.
 */
public class BundleActivator implements org.osgi.framework.BundleActivator {

  private ServiceRegistration<MetricRegistry> metricRegistry;
  private ServiceRegistration<HealthCheckRegistry> healthRegistry;

  @Override
  public void start(final BundleContext context) throws Exception {
    metricRegistry = context.registerService(MetricRegistry.class, new MetricRegistry(), new Hashtable<String, Object>());
    healthRegistry = context.registerService(HealthCheckRegistry.class, new HealthCheckRegistry(), new Hashtable<String, Object>());
  }

  @Override
  public void stop(final BundleContext context) throws Exception {
    context.ungetService(metricRegistry.getReference());
    context.ungetService(healthRegistry.getReference());
  }

}
