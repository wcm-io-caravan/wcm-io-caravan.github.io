20  package io.wcm.caravan.io.http.response;
21  
22  import static com.google.common.base.Preconditions.checkNotNull;
23  import static com.google.common.base.Preconditions.checkState;
24  
25  import java.util.LinkedList;
26  import java.util.List;
27  import java.util.Map;
28  
29  import org.osgi.annotation.versioning.ProviderType;
30  
31  import com.google.common.base.Joiner;
32  import com.google.common.collect.ImmutableMultimap;
33  import com.google.common.collect.LinkedHashMultimap;
34  import com.google.common.collect.Multimap;
35  
36  import io.wcm.caravan.io.http.impl.CaravanHttpHelper;
37  
38  /**
39   * An immutable response to an http invocation which only returns string content.
40   */
41  @ProviderType
42  public final class CaravanHttpResponse {
43  
44    private final int status;
45    private final String reason;
46    private final Multimap<String, String> headers;
47    private final Body body;
48  
49    CaravanHttpResponse(int status, String reason, Multimap<String, String> headers, Body body) {
50      checkState(status >= 200, "Invalid status code: %s", status);
51      this.status = status;
52      this.reason = checkNotNull(reason, "reason");
53      Multimap<String, String> copyOf = LinkedHashMultimap.create(checkNotNull(headers, "headers"));
54      this.headers = ImmutableMultimap.copyOf(copyOf);
55      this.body = body; // nullable
56    }
57  
58    /**
59     * status code. ex {@code 200} See <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec10.html" >rfc2616</a>
60     * @return HTTP status code
61     */
62    public int status() {
63      return status;
64    }
65  
66    /**
67     * @return HTTP status reason
68     */
69    public String reason() {
70      return reason;
71    }
72  
73    /**
74     * @return HTTP headers
75     */
76    public Multimap<String, String> headers() {
77      return headers;
78    }
79  
80    /**
81     * Collects all "Cache-Control" directives from the response headers into a single map. The keys in the map are the
82     * directive names (e.g. "max-age", "no-cache"), and everything after the "=" is taken as value. For directives that
83     * don't have a value "true" is used as a value instead.
84     * @return the map of Cache-Control directice
85     */
86    public Map<String, String> getCacheControl() {
87  
88      // http headers are case-insensitive, and even if we write all our headers in upper-case,
89      // any proxy in-between can turn them into lower-case variants.
90      // We need to make sure that we don't miss the lower-case variants here
91      List<String> cacheControlHeaders = new LinkedList<>();
92      cacheControlHeaders.addAll(headers.get("Cache-Control"));
93      cacheControlHeaders.addAll(headers.get("cache-control"));
94  
95      return CaravanHttpHelper.convertMultiValueHeaderToMap(cacheControlHeaders);
96    }
97  
98    /**
99     * if present, the response had a body
100    * @return HTTP body
101    */
102   public Body body() {
103     return body;
104   }
105 
106   @Override
107   public String toString() {
108     StringBuilder builder = new StringBuilder();
109     builder.append("HTTP/1.1 ").append(status).append(' ').append(reason).append('\n');
110     for (String field : headers.keySet()) {
111       builder.append(field).append(": ").append(Joiner.on(", ").join(headers.get(field))).append('\n');
112     }
113     if (body != null) {
114       builder.append('\n').append(body);
115     }
116     return builder.toString();
117   }
118 
119 }
