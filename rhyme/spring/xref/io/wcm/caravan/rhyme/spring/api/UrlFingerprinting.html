20  package io.wcm.caravan.rhyme.spring.api;
21  
22  import java.time.Duration;
23  import java.time.Instant;
24  import java.util.function.Supplier;
25  
26  import org.springframework.hateoas.server.mvc.WebMvcLinkBuilder;
27  import org.springframework.web.context.annotation.RequestScope;
28  
29  
30  /**
31   * Adds support for URL fingerprinting to your server-side resource implementations, by ensuring
32   * that all links in your service are using URLs that are guaranteed to change whenever the
33   * data, code or configuration of your service is updated. Responses for any URL that contains such a fingerprint
34   * can be considered immutable and cached indefinitely by the clients. If data (or code) changes, the clients will still
35   * pick up the most version of the data because the API entry point will point to different URLs with a new fingerprint.
36   * <p>
37   * To enable URL fingerprinting in your service, you need to do the following:
38   * </p>
39   * <ol>
40   * <li>Define a central location in your code where you call {@link SpringRhyme#enableUrlFingerprinting()} to create
41   * and configure this instance. In a Spring application, a {@link RequestScope} component works well as long as you
42   * stick to
43   * single-threaded request processing</li>
44   * <li>Call {@link #withTimestampParameter(String, Supplier)} to define the name of a
45   * query parameter to be used for URL fingerprinting, and the source for a last modification date</li>
46   * <li>Call {@link #withConditionalMaxAge(Duration, Duration)} to enable modification
47   * of your response's max-age cache-control directive depending on the presence of the URL fingerprint</li>
48   * <li>Ensure that all the links between your server-side resoures are then created using
49   * {@link #createLinkWith(WebMvcLinkBuilder)} of the same configured {@link UrlFingerprinting} instance</li>
50   * </ol>
51   */
52  public interface UrlFingerprinting {
53  
54    /**
55     * Add an URL fingerprinting parameter based on a last modification date
56     * @param name of the query parameter to be added
57     * @param source a function to retrieve the value of this parameter if it is not already present in the incoming
58     *          request
59     * @return this instance
60     */
61    UrlFingerprinting withTimestampParameter(String name, Supplier<Instant> source);
62  
63    /**
64     * Allows the URL fingerprinting logic to modify the max-age cache-control directive for the response that is
65     * currently rendered, depending on whether the incoming request is using a fingerprinted URL.
66     * @param mutableMaxAge to use for the entry point (or any other resources without a fingerprint in the URL).
67     * @param immutableMaxAge to use for all resources that do contain a fingerprint in their URL.
68     * @return this instance
69     */
70    UrlFingerprinting withConditionalMaxAge(Duration mutableMaxAge, Duration immutableMaxAge);
71  
72    /**
73     * Determines if the current incoming request is using URL fingerprinting.
74     * @return true if the query contains values for all timestamp parameters defined with
75     *         {@link #withTimestampParameter(String, Supplier)}
76     */
77    boolean isUsedInIncomingRequest();
78  
79    /**
80     * Creates a {@link RhymeLinkBuilder} that generates the URL to a controller method and adds the
81     * additional URL fingerprinting parameters
82     * @param linkBuilder created with {@link WebMvcLinkBuilder#linkTo(Class)} and
83     *          {@link WebMvcLinkBuilder#methodOn(Class, Object...)}
84     * @return a {@link RhymeLinkBuilder} to decorate the link with name and title attributes, and
85     *         then finally build it
86     */
87    RhymeLinkBuilder createLinkWith(WebMvcLinkBuilder linkBuilder);
88  
89  }
