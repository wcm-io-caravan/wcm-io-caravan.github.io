20  package io.wcm.caravan.rhyme.jaxrs.api;
21  
22  import java.util.Map;
23  import java.util.function.Consumer;
24  
25  import javax.ws.rs.core.Application;
26  
27  import org.osgi.annotation.versioning.ProviderType;
28  import org.osgi.service.component.annotations.Component;
29  import org.osgi.service.jaxrs.whiteboard.propertytypes.JaxrsResource;
30  
31  import io.wcm.caravan.hal.resource.Link;
32  import io.wcm.caravan.rhyme.jaxrs.impl.JaxRsControllerProxyLinkBuilder;
33  
34  /**
35   * @param <JaxRsResourceType> the class of the {@link Component} annotated with {@link JaxrsResource}
36   */
37  @ProviderType
38  public interface JaxRsLinkBuilder<JaxRsResourceType> {
39  
40    /**
41     * @param parameters to append to the parameters
42     * @return this
43     */
44    JaxRsLinkBuilder<JaxRsResourceType> withAdditionalQueryParameters(Map<String, Object> parameters);
45  
46    /**
47     * @param consumer a function to be called on the proxy that fills in the parameters for the resource
48     * @return a {@link Link} instance with the href already set
49     */
50    Link buildLinkTo(Consumer<JaxRsResourceType> consumer);
51  
52    /**
53     * Factory method to create {@link JaxRsLinkBuilder} instances
54     * @param baseUrl the base path of the JAX-RS {@link Application}
55     * @param resourceClass the class of the {@link Component} annotated with {@link JaxrsResource}
56     * @return the created instance
57     * @param <JaxRsResourceType> the type of the resource class
58     */
59    static <JaxRsResourceType> JaxRsLinkBuilder<JaxRsResourceType> create(String baseUrl, Class<JaxRsResourceType> resourceClass) {
60  
61      return new JaxRsControllerProxyLinkBuilder<JaxRsResourceType>(baseUrl, resourceClass);
62    }
63  
64  }
