13  /**
14   * An abstract implementation of {@link LinkableResource}, that you can use as a super class
15   * for your sling models implementing your {@link HalApiInterface}s. It provides injected fields with
16   * the commonly used {@link SlingRhyme} and {@link SlingResourceAdapter} sling models, and
17   * also contains an implementation of {@link LinkableResource#createLink()} that uses
18   * the {@link SlingLinkBuilder} to generate the link.
19   */
20  public abstract class AbstractLinkableResource implements LinkableResource, SlingLinkableResource {
21  
22    @Self
23    protected SlingRhyme rhyme;
24  
25    @Self
26    protected SlingResourceAdapter resourceAdapter;
27  
28    @Self
29    protected SlingLinkBuilder linkBuilder;
30  
31    private final LinkProperties linkProperties = new LinkProperties();
32  
33    @Override
34    public Link createLink() {
35  
36      if (linkProperties.getTitle() == null) {
37        linkProperties.setTitle(getDefaultLinkTitle());
38      }
39  
40      if (linkProperties.getName() == null) {
41        linkProperties.setName(rhyme.getCurrentResource().getName());
42      }
43  
44      return linkBuilder.createLinkToCurrentResource(this);
45    }
46  
47    @Override
48    public LinkProperties getLinkProperties() {
49  
50      return linkProperties;
51    }
52  
53    protected abstract String getDefaultLinkTitle();
54  
55  }
