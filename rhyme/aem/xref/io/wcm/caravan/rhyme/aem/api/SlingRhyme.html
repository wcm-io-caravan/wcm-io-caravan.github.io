40  /**
41   * An AEM-specific wrapper for the general {@link Rhyme} facade interface.
42   * <p>
43   * It is implemented as a sling model that is initially adapted from the incoming request to
44   * the framework's central {@link HalApiServlet} and will internally create a single {@link Rhyme} instance
45   * that will be used for all interaction with the core framework for the duration of the request.
46   * </p>
47   * <p>
48   * Your should implement your Rhyme resources as a
49   * <a href="https://sling.apache.org/documentation/bundles/models.html">Sling Model</a> class that adapts from
50   * {@link SlingRhyme} to your interface annotated with {@link HalApiInterface}, e.g.
51   * </p>
52   * <code>
53   * &#64;Model(adaptables = SlingRhyme.class, adapters = YourHalApiInterface.class)
54   * </code>
55   * <p>
56   * The {@link SlingRhyme} instance can be adapted to {@link SlingHttpServletRequest} and {@link Resource}, but also
57   * to any other types (e.g. sling models, or AEM {@link Page}) that are adaptable from those types. This allows you to
58   * to use the {@link Self} annotation to inject any any of those types into your model,
59   * and they will be resolved with the context of the resource that is currently being processed.
60   * </p>
61   * <p>
62   * This context resource will initially be set to the resource that was originally requested. You can use
63   * {@link #adaptResource(Resource, Class)} to create other sling models for related
64   * resources that you want to render with Rhyme. When calling that method, a new (lightweight) {@link SlingRhyme}
65   * instance will be created that is using the new given resource as a context for resource-based injection (but is still
66   * backed by the original {@link Rhyme} instance and the incoming request).
67   * </p>
68   * @see SlingResourceAdapter
69   * @see AbstractLinkableResource
70   * @see RhymeResourceRegistration
71   */
72  public interface SlingRhyme extends Adaptable {
73  
74    /**
75     * @return the current context resource being processed
76     */
77    @NotNull
78    Resource getCurrentResource();
79  
80    /**
81     * @return the resource that was originally requested in the incoming HTTP request
82     */
83    @NotNull
84    Resource getRequestedResource();
85  
86    /**
87     * Creates a new sling model instance for the given resource. Internally this will create a new {@link SlingRhyme}
88     * instance with the new resource context from which the model class will be adapted.
89     * @param resource from which the model will be adapted
90     * @param slingModelClass that is adaptable from {@link SlingRhyme}
91     * @param <T> the sling model class
92     * @return a new sling model instance for which all fields have been initialized
93     * @throws HalApiDeveloperException if the resource could not be adapted
94     */
95    <T> @NotNull T adaptResource(Resource resource, Class<T> slingModelClass);
96  
97    /**
98     * Limit the maximum time for which the response should be cached by clients and downstream services. Note that
99     * calling this method only sets the upper limit: if other upstream resource fetched during the current request
100    * indicate a lower max-age value in their header, that lower value will be used instead.
101    * @param duration the max cache time
102    */
103   void setResponseMaxAge(Duration duration);
104 
105   /**
106    * Create a dynamic client proxy to load HAL+JSON resources from an upstream service.
107    * @param <T> an interface annotated with {@link HalApiInterface}
108    * @param uri the URI of the entry point, in any format that the {@link HalResourceLoader} being used can understand
109    * @param halApiInterface an interface annotated with {@link HalApiInterface}
110    * @return a dynamic proxy instance of the provided {@link HalApiInterface} that you can use to navigate through the
111    *         resources of the service
112    */
113   <T> @NotNull T getRemoteResource(String uri, Class<T> halApiInterface);
114 
115   /**
116    * Provides access to the underlying {@link Rhyme} instance of the core framework, just in case you need
117    * to call one of the methods for which there doesn't exist a delegate in the {@link SlingRhyme} interface
118    * @return the single {@link Rhyme} instance that is used throughout the incoming request
119    */
120   Rhyme getCoreRhyme();
121 
122 }
