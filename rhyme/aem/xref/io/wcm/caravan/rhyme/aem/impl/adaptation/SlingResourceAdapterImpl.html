64    private SlingResourceAdapter fromResource(Resource resource) {
65  
66      if (fromResource != null) {
67        throw new HalApiDeveloperException(
68            "You cannot call the from* methods multiple times, as a single context resource is required for the following selection steps");
69      }
70  
71      if (resourceSelector.resources != null) {
72        throw new HalApiDeveloperException("The SlingResourceAdapterImpl#fromXyz methods must be called *before* any of the selectXyz methods are called");
73      }
74  
75      return new SlingResourceAdapterImpl(this, resource);
76    }

199   @Override
200   public SlingResourceAdapter selectResourceAt(String path) {
201 
202     if (path == null) {
203       return new SlingResourceAdapterImpl(this, null);
204     }
205 
206     Resource resource = slingRhyme.getRequestedResource().getResourceResolver().getResource(path);
207     if (resource == null) {
208       return resourceSelector.add(Stream.empty(), "non-existent resource at " + path);
209     }
210 
211     return resourceSelector.add(Stream.of(resource), "different resource at " + resource.getPath());
212   }

228   @Override
229   public <T> SlingResourceAdapter filterAdaptableTo(Class<T> adapterClazz, Predicate<T> predicate) {
230 
231     Predicate<Resource> isAdaptableAndAcceptedByPredicated = (res) -> {
232       T adapted = res.adaptTo(adapterClazz);
233       if (adapted == null) {
234         return false;
235       }
236       return predicate.test(adapted);
237     };
238 
239     return resourceFilter.add(isAdaptableAndAcceptedByPredicated,
240         "if resource can be adapted to " + adapterClazz.getSimpleName() + " and is accepted by predicate " + predicate);
241   }

287   private final class ResourceFilter {
288 
289     private final String description;
290     private final Predicate<Resource> predicate;
291 
292     private ResourceFilter(String description, Predicate<Resource> predicate) {
293       this.description = description;
294       this.predicate = predicate;
295     }
296 
297     private SlingResourceAdapter add(Predicate<Resource> newPredicate, String newDescription) {
298 
299       ResourceFilter newFilter;
300       if (predicate != null) {
301 
302         String mergedDescription = this.description + " and " + newDescription;
303         Predicate<Resource> mergedPredicate = predicate.and(newPredicate);
304 
305         newFilter = new ResourceFilter(mergedDescription, mergedPredicate);
306       }
307       else {
308         newFilter = new ResourceFilter(newDescription, newPredicate);
309       }
310 
311       return new SlingResourceAdapterImpl(SlingResourceAdapterImpl.this, resourceSelector, newFilter);
312     }
313 
314   }

316   private final class ResourceSelector {
317 
318     private final String description;
319     private final Stream<Resource> resources;
320 
321     private ResourceSelector(String description, Stream<Resource> resources) {
322       this.description = description;
323       this.resources = resources;
324     }
325 
326     private SlingResourceAdapter add(Function<Resource, Stream<Resource>> streamFunc, String newDescription) {
327 
328       Resource contextResource = fromResource;
329       if (contextResource == null) {
330         contextResource = slingRhyme.getCurrentResource();
331       }
332 
333       Stream<Resource> stream = streamFunc.apply(contextResource);
334 
335       return add(stream, newDescription.replace("{}", contextResource.getPath()));
336     }
337 
338     private SlingResourceAdapter add(@NotNull Stream<Resource> newResources, String newDescription) {
339 
340       Preconditions.checkNotNull(newResources, "the stream of resources must not be null");
341 
342       ResourceSelector newSelector;
343       if (resources != null) {
344 
345         String mergedDescription = this.description + " and " + newDescription;
346         Stream<Resource> mergedStream = Stream.concat(this.resources, newResources);
347 
348         newSelector = new ResourceSelector(mergedDescription, mergedStream);
349       }
350       else {
351         newSelector = new ResourceSelector(newDescription, newResources);
352       }
353 
354 
355       return new SlingResourceAdapterImpl(SlingResourceAdapterImpl.this, newSelector, resourceFilter);
356     }
357   }
