20  package io.wcm.caravan.rhyme.aem.impl.resources;
21  
22  import java.time.Duration;
23  import java.util.List;
24  import java.util.stream.Collectors;
25  
26  import javax.inject.Inject;
27  
28  import org.apache.sling.models.annotations.Model;
29  
30  import io.wcm.caravan.hal.resource.Link;
31  import io.wcm.caravan.rhyme.aem.api.SlingRhyme;
32  import io.wcm.caravan.rhyme.aem.api.resources.AbstractLinkableResource;
33  import io.wcm.caravan.rhyme.aem.impl.RhymeResourceRegistry;
34  import io.wcm.caravan.rhyme.api.resources.LinkableResource;
35  
36  @Model(adaptables = SlingRhyme.class, adapters = AemApiDiscoveryResource.class)
37  public class AemApiDiscoveryResourceImpl extends AbstractLinkableResource implements AemApiDiscoveryResource {
38  
39    public static final Integer MAX_AGE_SECONDS = 60;
40  
41    @Inject
42    private RhymeResourceRegistry registry;
43  
44    @Override
45    public List<Link> getApiEntryPoints() {
46  
47      rhyme.setResponseMaxAge(Duration.ofSeconds(MAX_AGE_SECONDS));
48  
49      return registry.getAllApiEntryPoints(resourceAdapter)
50          .map(LinkableResource::createLink)
51          .collect(Collectors.toList());
52    }
53  
54    @Override
55    protected String getDefaultLinkTitle() {
56      return "A list of all HAL+JSON APIs registered on this AEM instance";
57    }
58  
59  }
