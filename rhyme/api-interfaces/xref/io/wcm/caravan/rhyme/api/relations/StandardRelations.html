21  package io.wcm.caravan.rhyme.api.relations;
22  
23  import io.wcm.caravan.rhyme.api.annotations.Related;
24  
25  /**
26   * Constants for some standard link relations used in {@link Related} annotations.
27   * @see <a href="http://www.iana.org/assignments/link-relations/link-relations.xhtml">IANA Link Relation Types</a>
28   */
29  public final class StandardRelations {
30  
31    private StandardRelations() {
32      // constants only
33    }
34  
35    /**
36     * Refers to a resource that is the subject of the link's context.
37     * @see <a href="https://www.iana.org/go/rfc6903">RFC6903, section 2</a>
38     */
39    public static final String ABOUT = "about";
40  
41    /**
42     * Refers to a substitute for this context
43     * @see <a href="http://www.w3.org/TR/html5/links.html#link-type-alternate">HTML5 link-type-alternate</a>
44     */
45    public static final String ALTERNATE = "alternate";
46  
47    /**
48     * Designates the preferred version of a resource (the IRI and its contents).
49     * @see <a href="http://www.iana.org/go/rfc6596">RFC6596</a>
50     */
51    public static final String CANONICAL = "canonical";
52  
53    /**
54     * The target IRI points to a resource which represents the collection resource for the context IRI.
55     * @see <a href="http://www.iana.org/go/rfc6573">RFC6573</a>
56     */
57    public static final String COLLECTION = "collection";
58  
59    /**
60     * A reserved link relation which you can use to hint at the location of resource documentation.
61     * There can be multiple links in the 'curies' section. They come with a 'name' and a templated 'href' which must
62     * contain the {rel} placeholder.
63     * To retrieve documentation about the latest-posts resource, the client will expand the associated CURIE link with
64     * the actual link's 'rel'
65     * @see <a href="https://datatracker.ietf.org/doc/html/draft-kelly-json-hal#section-8.2">draft-kelly-json-hal-08</a>
66     */
67    public static final String CURIES = "curies";
68  
69    /**
70     * Refers to a resource that is not part of the same site as the current context.
71     * @see <a href="https://html.spec.whatwg.org/multipage/links.html#link-type-external">HTML Spec</a>
72     */
73    public static final String EXTERNAL = "external";
74  
75    /**
76     * An IRI that refers to the furthest preceding resource in a series of resources.
77     * @see <a href="http://www.iana.org/go/rfc5988">RFC5988</a>
78     */
79    public static final String FIRST = "first";
80  
81    /**
82     * Refers to an index.
83     * @see <a href="http://www.w3.org/TR/1999/REC-html401-19991224">HTML 4.01 Specification</a>
84     */
85    public static final String INDEX = "index";
86  
87    /**
88     * The target IRI points to a resource that is a member of the collection represented by the context IRI.
89     * @see <a href="http://www.iana.org/go/rfc6573">RFC6573</a>
90     */
91    public static final String ITEM = "item";
92  
93    /**
94     * An IRI that refers to the furthest following resource in a series of resources.
95     * @see <a href="http://www.iana.org/go/rfc5988">RFC5988</a>
96     */
97    public static final String LAST = "last";
98  
99    /**
100    * The Target IRI points to a Memento, a fixed resource that will not change state anymore
101    * @see <a href="https://www.iana.org/go/rfc7089">RFC7089</a>
102    */
103   public static final String MEMENTO = "memento";
104 
105   /**
106    * Indicates that the link's context is a part of a series, and that the next in the series is the link target.
107    * @see <a href="http://www.w3.org/TR/html5/links.html#link-type-next">HTML5 Recommendation</a>
108    */
109   public static final String NEXT = "next";
110 
111   /**
112    * Indicates that the link's context is a part of a series, and that the previous in the series is the link target.
113    * @see <a href="http://www.w3.org/TR/html5/links.html#link-type-next">HTML5 Recommendation</a>
114    */
115   public static final String PREV = "prev";
116 
117   /**
118    * Identifies a related resource.
119    * @see <a href="http://www.iana.org/go/rfc4287">RFC4287</a>
120    */
121   public static final String RELATED = "related";
122 
123   /**
124    * Refers to a resource that can be used to search through the link's context and related resources.
125    * @see <a href="http://www.opensearch.org/Specifications/OpenSearch/1.1">OpenSearch</a>
126    */
127   public static final String SEARCH = "search";
128 
129   /**
130    * Refers to a section in a collection of resources.
131    * @see <a href="https://www.w3.org/TR/html401/">HTML 4.01 Specification</a>
132    */
133   public static final String SECTION = "section";
134 
135   /**
136    * Conveys an identifier for the link's context.
137    * @see <a href="http://www.iana.org/go/rfc4287">RFC4287</a>
138    */
139   public static final String SELF = "self";
140 
141   /**
142    * Refers to a resource serving as a subsection in a collection of resources.
143    * @see <a href="https://www.w3.org/TR/html401/">HTML 4.01 Specification</a>
144    */
145   public static final String SUBSECTION = "subsection";
146 
147   /**
148    * Refers to a parent document in a hierarchy of documents.
149    * @see <a href="http://www.iana.org/go/rfc5988">RFC5988</a>
150    */
151   public static final String UP = "up";
152 
153   /**
154    * Identifies a resource that is the source of the information in the link's context.
155    * @see <a href="http://www.iana.org/go/rfc4287">RFC4287</a>
156    */
157   public static final String VIA = "via";
158 
159 }
