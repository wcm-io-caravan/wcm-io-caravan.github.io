20  package io.wcm.caravan.rhyme.api.resources;
21  
22  import org.osgi.annotation.versioning.ConsumerType;
23  
24  import io.wcm.caravan.hal.resource.Link;
25  import io.wcm.caravan.rhyme.api.annotations.ResourceLink;
26  
27  /**
28   * An interface to be implemented by all resources that should be accessible directly via a http(s) URL. If you don't
29   * implement this interface in your server-side implementation of a HAL API interface, then that resource can only be
30   * embedded (and you must implement EmbeddableResource instead).
31   */
32  @ConsumerType
33  public interface LinkableResource {
34  
35    /**
36     * Create a link to this resource, including meaningful title and name properties where appropriate. If all required
37     * parameters of the resource are set, then the link should have a resolved URI as href property. If some or all
38     * required parameters are null, a link with a URI template should be created instead.
39     * @return a {@link Link} instance where href, title and name properties are already set as required
40     */
41    @ResourceLink
42    Link createLink();
43  }
