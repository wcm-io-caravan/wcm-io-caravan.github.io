54  /**
55   * An abstract base class for the common customization and dependency injection required by the {@link RhymeBuilder},
56   * {@link HalApiClientBuilder} and {@link HalResponseRendererBuilder} implementations.
57   * @param <BuilderInterface> the interface that the subclass is implementing
58   */
59  abstract class AbstractRhymeBuilder<BuilderInterface> {
60  
61    private final Stopwatch stopwatch = Stopwatch.createStarted();
62  
63    private HalResourceLoader resourceLoader;
64  
65    private RequestMetricsCollector metrics;
66  
67    private final List<ExceptionStatusAndLoggingStrategy> exceptionStrategies = new ArrayList<>();
68  
69    private final List<HalApiTypeSupport> typeSupports = new ArrayList<>();
70  
71    private RhymeDocsSupport rhymeDocsSupport;
72  
73    protected boolean wasUsedToBuild;
74  
75    @SuppressWarnings("unchecked")
76    public BuilderInterface withResourceLoader(HalResourceLoader resourceLoader) {
77  
78      this.resourceLoader = resourceLoader;
79      return (BuilderInterface)this;
80    }
81  
82    @SuppressWarnings("unchecked")
83    public BuilderInterface withMetrics(RequestMetricsCollector sharedMetrics) {
84  
85      this.metrics = sharedMetrics;
86      return (BuilderInterface)this;
87    }
88  
89    @SuppressWarnings("unchecked")
90    public BuilderInterface withRhymeDocsSupport(RhymeDocsSupport rhymeDocsSupport) {
91  
92      this.rhymeDocsSupport = rhymeDocsSupport;
93      return (BuilderInterface)this;
94    }
95  
96    @SuppressWarnings("unchecked")
97    public BuilderInterface withReturnTypeSupport(HalApiReturnTypeSupport additionalTypeSupport) {
98  
99      if (additionalTypeSupport != null) {
100       typeSupports.add(new HalApiTypeSupportAdapter(additionalTypeSupport));
101     }
102     return (BuilderInterface)this;
103   }
104 
105   @SuppressWarnings("unchecked")
106   public BuilderInterface withAnnotationTypeSupport(HalApiAnnotationSupport additionalTypeSupport) {
107 
108     if (additionalTypeSupport != null) {
109       typeSupports.add(new HalApiTypeSupportAdapter(additionalTypeSupport));
110     }
111     return (BuilderInterface)this;
112   }
113 
114   @SuppressWarnings("unchecked")
115   public BuilderInterface withExceptionStrategy(ExceptionStatusAndLoggingStrategy customStrategy) {
116 
117     exceptionStrategies.add(customStrategy);
118     return (BuilderInterface)this;
119   }
120 
121   private ExceptionStatusAndLoggingStrategy getEffectiveExceptionStrategy() {
122 
123     List<ExceptionStatusAndLoggingStrategy> nonNullStrategies = exceptionStrategies.stream()
124         .filter(Objects::nonNull)
125         .collect(Collectors.toList());
126 
127     if (nonNullStrategies.isEmpty()) {
128       return null;
129     }
130     if (nonNullStrategies.size() == 1) {
131       return nonNullStrategies.get(0);
132     }
133 
134     return new CompositeExceptionStatusAndLoggingStrategy(nonNullStrategies);
135   }
136 
137   private HalApiTypeSupport getEffectiveTypeSupport() {
138 
139     DefaultHalApiTypeSupport defaultSupport = new DefaultHalApiTypeSupport();
140 
141     if (typeSupports.isEmpty()) {
142       return defaultSupport;
143     }
144 
145     List<HalApiTypeSupport> customAndDefault = new ArrayList<>(typeSupports);
146     customAndDefault.add(defaultSupport);
147 
148     return new CompositeHalApiTypeSupport(customAndDefault);
149   }
150 
151   private void applyDefaultsBeforeBuilding() {
152 
153     if (wasUsedToBuild) {
154       throw new HalApiDeveloperException("You shouldn't re-use this builder to create more than once instance");
155     }
156 
157     if (resourceLoader == null) {
158       resourceLoader = HalResourceLoader.create();
159     }
160 
161     if (metrics == null) {
162       metrics = RequestMetricsCollector.create();
163     }
164   }
165 
166   AsyncHalResponseRenderer buildAsyncRenderer() {
167 
168     applyDefaultsBeforeBuilding();
169 
170     HalApiTypeSupport typeSupport = getEffectiveTypeSupport();
171 
172     AsyncHalResourceRenderer resourceRenderer = new AsyncHalResourceRendererImpl(metrics, typeSupport);
173 
174     ExceptionStatusAndLoggingStrategy exceptionStrategy = getEffectiveExceptionStrategy();
175 
176     return new AsyncHalResponseRendererImpl(resourceRenderer, metrics, exceptionStrategy, typeSupport, rhymeDocsSupport);
177   }
178 
179   HalApiClient buildApiClient() {
180 
181     applyDefaultsBeforeBuilding();
182 
183     HalApiTypeSupport effectiveTypeSupport = getEffectiveTypeSupport();
184 
185     return new HalApiClientImpl(resourceLoader, metrics, effectiveTypeSupport);
186   }
187 
188   @SuppressWarnings("deprecation")
189   Rhyme buildRhyme(String incomingRequestUri) {
190 
191     HalApiClient client = buildApiClient();
192 
193     AsyncHalResponseRenderer renderer = buildAsyncRenderer();
194 
195     VndErrorResponseRenderer errorRenderer = VndErrorResponseRenderer.create(getEffectiveExceptionStrategy());
196 
197     RhymeImpl impl = new RhymeImpl(incomingRequestUri, client, renderer, errorRenderer, metrics);
198 
199     metrics.onMethodInvocationFinished(AsyncHalResponseRenderer.class, "building Rhyme instance", stopwatch.elapsed(TimeUnit.MICROSECONDS));
200 
201     return impl;
202   }
203 }
