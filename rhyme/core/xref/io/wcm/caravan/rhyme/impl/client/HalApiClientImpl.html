20  package io.wcm.caravan.rhyme.impl.client;
21  
22  import com.google.common.base.Preconditions;
23  
24  import io.wcm.caravan.rhyme.api.client.HalApiClient;
25  import io.wcm.caravan.rhyme.api.common.RequestMetricsCollector;
26  import io.wcm.caravan.rhyme.api.spi.HalResourceLoader;
27  import io.wcm.caravan.rhyme.impl.client.proxy.HalApiClientProxyFactory;
28  import io.wcm.caravan.rhyme.impl.reflection.HalApiTypeSupport;
29  
30  /**
31   * The implementation of {@link HalApiClient} that delegates the actual loading of resources via the
32   * {@link HalResourceLoader} interface to the {@link HalResourceLoaderWrapper}, and creates dynamic
33   * client proxies with the {@link HalApiClientProxyFactory}.
34   */
35  public class HalApiClientImpl implements HalApiClient {
36  
37    private final HalApiClientProxyFactory factory;
38  
39    private final HalApiTypeSupport typeSupport;
40  
41    /**
42     * @param resourceLoader implements the actual loading (and caching) of JSON/HAL resources via any HTTP client library
43     * @param metrics an instance of {@link RequestMetricsCollector} to collect performance relevant data for the current
44     *          incoming request
45     * @param typeSupport the strategy to detect HAL API annotations and perform type conversions
46     */
47    public HalApiClientImpl(HalResourceLoader resourceLoader, RequestMetricsCollector metrics, HalApiTypeSupport typeSupport) {
48  
49      Preconditions.checkNotNull(resourceLoader, "A " + HalResourceLoader.class.getName() + " instance must be provided");
50      HalResourceLoaderWrapper wrapper = new HalResourceLoaderWrapper(resourceLoader, metrics);
51  
52      factory = new HalApiClientProxyFactory(wrapper, metrics, typeSupport);
53  
54      this.typeSupport = typeSupport;
55    }
56  
57    @Override
58    public <T> T getRemoteResource(String uri, Class<T> halApiInterface) {
59  
60      // create a proxy instance that loads the entry point lazily when required by any method call on the proxy
61      return factory.createProxyFromUrl(halApiInterface, uri);
62    }
63  
64    public HalApiTypeSupport getTypeSupport() {
65      return typeSupport;
66    }
67  }
