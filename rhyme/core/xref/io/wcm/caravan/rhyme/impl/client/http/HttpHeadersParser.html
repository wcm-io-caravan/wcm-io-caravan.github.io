3   import java.time.Duration;
4   import java.util.Collection;
5   import java.util.Map;
6   import java.util.Objects;
7   import java.util.Optional;
8   import java.util.regex.Matcher;
9   import java.util.regex.Pattern;
10  import java.util.stream.Stream;
11  
12  import org.apache.commons.lang3.StringUtils;
13  
14  import com.google.common.collect.Ordering;
15  
16  import io.wcm.caravan.rhyme.api.spi.HttpClientCallback;
17  
18  /**
19   * Extracts the relevant information from the HTTP headers given to
20   * {@link HttpClientCallback#onHeadersAvailable(int, Map)}
21   */
22  class HttpHeadersParser {
23  
24    private static final String MAX_AGE_REGEX = ".*max-age=([0-9]+).*";
25  
26    private static final Pattern MAX_AGE_PATTERN = Pattern.compile(MAX_AGE_REGEX);
27  
28    private final Map<String, ? extends Collection<String>> headers;
29  
30    HttpHeadersParser(Map<String, ? extends Collection<String>> headers) {
31      this.headers = headers;
32    }
33  
34    Optional<String> getContentType() {
35  
36      return findHeader("content-type");
37    }
38  
39    Optional<Integer> getMaxAge() {
40  
41      return findHeaders("cache-control")
42          .map(HttpHeadersParser::parseMaxAge)
43          .filter(Objects::nonNull)
44          .max(Ordering.natural());
45    }
46  
47    private Optional<String> findHeader(String name) {
48  
49      return findHeaders(name)
50          .findFirst();
51    }
52  
53    Stream<String> findHeaders(String name) {
54      return headers.entrySet().stream()
55          .filter(entry -> StringUtils.equalsIgnoreCase(name, entry.getKey()))
56          .flatMap(entry -> entry.getValue().stream());
57    }
58  
59    private static Integer parseMaxAge(String cacheControl) {
60  
61      if (cacheControl.contains("immutable")) {
62        return (int)Duration.ofDays(365).getSeconds();
63      }
64  
65      Matcher matcher = MAX_AGE_PATTERN.matcher(cacheControl);
66      if (!matcher.matches()) {
67        return null;
68      }
69  
70      long maxAge = Long.parseLong(matcher.group(1));
71      if (maxAge > Integer.MAX_VALUE) {
72        return Integer.MAX_VALUE;
73      }
74      return (int)maxAge;
75    }
76  
77  }
