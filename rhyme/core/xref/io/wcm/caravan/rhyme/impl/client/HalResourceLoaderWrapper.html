20  package io.wcm.caravan.rhyme.impl.client;
21  
22  import java.util.concurrent.ExecutionException;
23  import java.util.concurrent.TimeUnit;
24  
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import com.google.common.base.Stopwatch;
29  import com.google.common.cache.Cache;
30  import com.google.common.cache.CacheBuilder;
31  import com.google.common.util.concurrent.UncheckedExecutionException;
32  
33  import io.reactivex.rxjava3.core.Single;
34  import io.wcm.caravan.hal.resource.HalResource;
35  import io.wcm.caravan.hal.resource.Link;
36  import io.wcm.caravan.rhyme.api.common.HalResponse;
37  import io.wcm.caravan.rhyme.api.common.RequestMetricsCollector;
38  import io.wcm.caravan.rhyme.api.exceptions.HalApiClientException;
39  import io.wcm.caravan.rhyme.api.exceptions.HalApiDeveloperException;
40  import io.wcm.caravan.rhyme.api.spi.HalResourceLoader;
41  import io.wcm.caravan.rhyme.impl.util.RxJavaTransformers;
42  
43  /**
44   * Extends another {@link HalResourceLoader} with common error handling and metrics collection functionality
45   */
46  class HalResourceLoaderWrapper implements HalResourceLoader {
47  
48    private static final Logger log = LoggerFactory.getLogger(HalResourceLoaderWrapper.class);
49  
50    private final Cache<String, Single<HalResponse>> cache = CacheBuilder.newBuilder().build();
51  
52    private final HalResourceLoader delegate;
53    private final RequestMetricsCollector metrics;
54  
55    HalResourceLoaderWrapper(HalResourceLoader delegate, RequestMetricsCollector metrics) {
56      this.delegate = delegate;
57      this.metrics = metrics;
58    }
59  
60    @Override
61    @SuppressWarnings("PMD.PreserveStackTrace")
62    public Single<HalResponse> getHalResource(String uri) {
63      try {
64        return cache.get(uri, () -> {
65  
66          Stopwatch stopwatch = Stopwatch.createUnstarted();
67  
68          Single<HalResponse> loadedResource = delegate.getHalResource(uri)
69              .doOnSubscribe(d -> startStopwatch(stopwatch))
70              .doOnError(ex -> registerErrorMetrics(uri, ex, stopwatch))
71              .doOnSuccess(jsonResponse -> registerResponseMetrics(uri, jsonResponse, stopwatch))
72              .onErrorResumeNext(ex -> rethrowUnexpectedExceptions(uri, ex));
73  
74          LinkRewriting rewriting = new LinkRewriting(uri);
75          Single<HalResponse> processedResource = loadedResource.map(rewriting::resolveRelativeLinks);
76  
77          Single<HalResponse> cachedResource = processedResource.compose(RxJavaTransformers.cacheSingleIfCompleted());
78  
79          return cachedResource;
80        });
81      }
82      catch (UncheckedExecutionException | ExecutionException ex) {
83        String msg = delegate.getClass() + "#getHalResource(String) returned null or threw an exception. "
84            + " Please make sure that your implementation will always return a Single instance during assembly time.";
85        throw new HalApiDeveloperException(msg, ex.getCause());
86      }
87    }
88  
89    private void startStopwatch(Stopwatch stopwatch) {
90      if (stopwatch.isRunning()) {
91        stopwatch.stop();
92        stopwatch.reset();
93      }
94      stopwatch.start();
95    }
96  
97    private void registerErrorMetrics(String uri, Throwable ex, Stopwatch stopwatch) {
98  
99      String title = "Upstream resource that failed to load: " + ex.getMessage();
100 
101     metrics.onResponseRetrieved(uri, title, null, stopwatch.elapsed(TimeUnit.MICROSECONDS));
102   }
103 
104   private void registerResponseMetrics(String uri, HalResponse jsonResponse, Stopwatch stopwatch) {
105 
106     log.debug("Received JSON response from {} with status code {} and max-age {} in {}ms",
107         uri, jsonResponse.getStatus(), jsonResponse.getMaxAge(), stopwatch.elapsed(TimeUnit.MILLISECONDS));
108 
109     String title = getResourceTitle(jsonResponse.getBody(), uri);
110 
111     metrics.onResponseRetrieved(uri, title, jsonResponse.getMaxAge(), stopwatch.elapsed(TimeUnit.MICROSECONDS));
112   }
113 
114   private Single<HalResponse> rethrowUnexpectedExceptions(String uri, Throwable ex) {
115 
116     if (ex instanceof HalApiClientException) {
117       return Single.error(ex);
118     }
119 
120     RuntimeException re = new HalApiDeveloperException("An unexpected exception was emitted by " + delegate.getClass().getName() + " when requesting " + uri
121         + ". Please make sure that your implementation rethrows all exceptions as HalApiClientException, to provide status code information whenever possible ",
122         ex);
123     return Single.error(re);
124   }
125 
126   private static String getResourceTitle(HalResource halResource, String uri) {
127 
128     Link selfLink = halResource.getLink();
129 
130     String title = null;
131     if (selfLink != null) {
132       title = selfLink.getTitle();
133     }
134 
135     if (title == null) {
136       title = "Untitled HAL resource from " + uri;
137     }
138 
139     return title;
140   }
