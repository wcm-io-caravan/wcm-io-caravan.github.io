46  /**
47   * Internal utility methods to invoke methods returning reactive streams, and converting between various
48   * reactive types.
49   */
50  public final class RxJavaReflectionUtils {
51  
52    private RxJavaReflectionUtils() {
53      // this class contains only static methods
54    }
55  
56    /**
57     * @param resourceImplInstance the object on which to invoke the method
58     * @param method a method that returns a {@link Single}, {@link Maybe}, {@link Observable} or {@link Publisher}
59     * @param metrics to track the method invocation time
60     * @param typeSupport the strategy to detect HAL API annotations and perform type conversions
61     * @return an {@link Observable} that emits the items from the reactive stream returned by the method
62     */
63    @SuppressWarnings("PMD.PreserveStackTrace")
64    public static Observable<?> invokeMethodAndReturnObservable(Object resourceImplInstance, Method method, RequestMetricsCollector metrics,
65        HalApiTypeSupport typeSupport) {
66  
67      String fullMethodName = HalApiReflectionUtils.getClassAndMethodName(resourceImplInstance, method, typeSupport);
68  
69      try (RequestMetricsStopwatch sw = metrics.startStopwatch(AsyncHalResourceRenderer.class, () -> "calls to " + fullMethodName)) {
70  
71        Object[] args = new Object[method.getParameterCount()];
72        Object returnValue = method.invoke(resourceImplInstance, args);
73  
74        if (returnValue == null) {
75          throw new HalApiDeveloperException(
76              fullMethodName + " must not return null. You should return an empty Maybe/Observable if the related resource does not exist");
77        }
78  
79        return convertToObservable(returnValue, typeSupport);
80      }
81      catch (InvocationTargetException ex) {
82        Throwable cause = ex.getTargetException();
83        if (cause instanceof RuntimeException) {
84          throw ((RuntimeException)cause);
85        }
86        throw new HalApiServerException(500, "A checked exception was thrown when calling " + fullMethodName, cause);
87      }
88      catch (IllegalAccessException | IllegalArgumentException ex) {
89        throw new HalApiDeveloperException("Failed to invoke method " + fullMethodName, ex);
90      }
91    }
92  
93    /**
94     * @param method a method that returns a Observable
95     * @return the type of the emitted results
96     */
97    public static Class<?> getObservableEmissionType(Method method, HalApiTypeSupport typeSupport) {
98      Type returnType = method.getGenericReturnType();
99  
100     if (!(returnType instanceof ParameterizedType)) {
101       return method.getReturnType();
102     }
103 
104     ParameterizedType observableType = (ParameterizedType)returnType;
105 
106     Function<Object, Observable<?>> conversion = typeSupport.convertToObservable(method.getReturnType());
107 
108     if (conversion == null) {
109       throw new HalApiDeveloperException("The return type " + method.getReturnType().getSimpleName()
110           + " of method " + method.getDeclaringClass().getSimpleName() + "#" + method.getName() + " is not supported."
111           + " If you do want to use this return type, you can add support for it by using " + RhymeBuilder.class.getSimpleName() + "#withReturnTypeSupport");
112     }
113 
114     Type resourceType = observableType.getActualTypeArguments()[0];
115 
116     Preconditions.checkArgument(resourceType instanceof Class,
117         "return types must be generic class with Class type parameters (e.g. List<ObjectNode>), but found " + resourceType.getTypeName());
118 
119     return (Class)resourceType;
120   }
121 
122   /**
123    * @param reactiveInstance a {@link Single}, {@link Maybe}, {@link Observable} or {@link Publisher}
124    * @param targetType {@link Single}, {@link Maybe}, {@link Observable} or {@link Publisher} class
125    * @param metrics to collect emission times
126    * @param description for the metrics
127    * @param typeSupport the strategy to perform type conversions of return values
128    * @return an instance of the target type that will replay (and cache!) the items emitted by the given reactive
129    *         instance
130    */
131   public static Observable<?> convertAndCacheReactiveType(Object reactiveInstance, Class<?> targetType, RequestMetricsCollector metrics,
132       Supplier<String> description, HalApiReturnTypeSupport typeSupport) {
133 
134     Observable<?> observable = convertToObservable(reactiveInstance, typeSupport)
135         .compose(EmissionStopwatch.collectMetrics(description, metrics));
136 
137     // do not use Observable#cache() here, because we want consumers to be able to use Observable#retry()
138     Observable<?> cached = observable.compose(RxJavaTransformers.cacheIfCompleted());
139 
140     return cached;
141   }
142 
143   /**
144    * Converts the observable into any other type supported by the given {@link HalApiTypeSupport} instance
145    * @param observable to convert
146    * @param targetType to which the observable should be convert
147    * @param typeSupport provides the available conversion functions
148    * @return an object of the given target type
149    */
150   public static Object convertObservableTo(Observable<?> observable, Class<?> targetType, HalApiReturnTypeSupport typeSupport) {
151 
152     Preconditions.checkNotNull(targetType, "A target type must be provided");
153 
154     Function<Observable, ?> conversion = typeSupport.convertFromObservable(targetType);
155     if (conversion == null) {
156       throw new HalApiDeveloperException("The given target type of " + targetType.getName() + " is not a supported return type");
157     }
158 
159     return conversion.apply(observable);
160   }
161 
162   private static Observable<?> convertToObservable(Object reactiveInstance, HalApiReturnTypeSupport typeSupport) {
163 
164     Preconditions.checkNotNull(reactiveInstance, "Cannot convert null objects");
165 
166     Function<Object, Observable<?>> conversion = typeSupport.convertToObservable(reactiveInstance.getClass());
167     if (conversion == null) {
168       throw new HalApiDeveloperException("The given instance of " + reactiveInstance.getClass().getName() + " is not a supported return type");
169     }
170 
171     return conversion.apply(reactiveInstance);
172   }
