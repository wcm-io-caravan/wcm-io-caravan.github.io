20  package io.wcm.caravan.rhyme.impl.client.proxy;
21  
22  import java.util.function.Function;
23  
24  import com.fasterxml.jackson.databind.JsonNode;
25  import com.fasterxml.jackson.databind.node.ObjectNode;
26  
27  import io.reactivex.rxjava3.core.Observable;
28  import io.wcm.caravan.hal.resource.HalResource;
29  import io.wcm.caravan.rhyme.api.annotations.ResourceRepresentation;
30  import io.wcm.caravan.rhyme.api.exceptions.HalApiDeveloperException;
31  
32  class ResourceRepresentationHandler implements Function<HalResource, Observable<Object>> {
33  
34    private final HalApiMethodInvocation invocation;
35  
36    ResourceRepresentationHandler(HalApiMethodInvocation invocation) {
37      this.invocation = invocation;
38    }
39  
40    @Override
41    public Observable<Object> apply(HalResource resource) {
42  
43      Class<?> emissionType = invocation.getEmissionType();
44  
45      if (emissionType.isAssignableFrom(HalResource.class)) {
46        return Observable.just(resource);
47      }
48  
49      if (emissionType.isAssignableFrom(ObjectNode.class)) {
50        return Observable.just(resource.getModel());
51      }
52  
53      if (emissionType.isAssignableFrom(String.class)) {
54        return Observable.just(resource.getModel().toString());
55      }
56  
57      throw new HalApiDeveloperException(
58          "The method " + invocation + " annotated with @" + ResourceRepresentation.class.getSimpleName() + " must return "
59              + "a reactive type emitting either " + HalResource.class.getSimpleName() + ", " + JsonNode.class.getSimpleName()
60              + ", " + String.class.getSimpleName());
61    }
62  }
