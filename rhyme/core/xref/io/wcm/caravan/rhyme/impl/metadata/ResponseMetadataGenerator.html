61  /**
62   * Full implementation of {@link RequestMetricsCollector} that keeps track of all upstream resources that have been
63   * retrieved, and additional invocation/emission times to analyze the performance of a request
64   */
65  public class ResponseMetadataGenerator implements RequestMetricsCollector {
66  
67    private static final Logger log = LoggerFactory.getLogger(ResponseMetadataGenerator.class);
68  
69    private static final Map<TimeUnit, String> TIME_UNIT_ABBRS = ImmutableMap.of(
70        TimeUnit.MINUTES, "m",
71        TimeUnit.SECONDS, "s",
72        TimeUnit.MILLISECONDS, "ms",
73        TimeUnit.MICROSECONDS, "μs",
74        TimeUnit.NANOSECONDS, "ns");
75  
76    private final Stopwatch overalResponseTimeStopwatch = Stopwatch.createStarted();
77  
78    private final List<TimeMeasurement> inputMaxAgeSeconds = Collections.synchronizedList(new ArrayList<>());
79    private final List<TimeMeasurement> inputResponseTimes = Collections.synchronizedList(new ArrayList<>());
80    private final ListMultimap<String, TimeMeasurement> methodInvocationTimes = Multimaps.synchronizedListMultimap(LinkedListMultimap.create());
81  
82    private final List<Link> sourceLinks = Collections.synchronizedList(new ArrayList<>());
83  
84    private Integer maxAgeLimit;
85  
86    private final AtomicBoolean metadataWasRendered = new AtomicBoolean();
87  
88    @Override
89    public void onResponseRetrieved(String resourceUri, String resourceTitle, Integer maxAgeSeconds, long responseTimeMicros) {
90  
91      if (metadataWasRendered.get()) {
92        log.warn("Response from {} was retrieved after embedded metadata resource was rendered", resourceUri);
93        return;
94      }
95  
96      if (maxAgeSeconds != null) {
97        inputMaxAgeSeconds.add(new TimeMeasurement(resourceUri, maxAgeSeconds / 1.f, TimeUnit.SECONDS));
98      }
99      inputResponseTimes.add(new TimeMeasurement(resourceUri, responseTimeMicros / 1000.f, TimeUnit.MILLISECONDS));
100 
101     Link link = new Link(resourceUri);
102     link.setTitle(resourceTitle);
103     sourceLinks.add(link);
104   }
105 
106   @Override
107   public void onMethodInvocationFinished(Class category, String methodDescription, long invocationDurationMicros) {
108     methodInvocationTimes.put(category.getSimpleName(),
109         new TimeMeasurement(methodDescription, invocationDurationMicros / 1000.f, TimeUnit.MILLISECONDS));
110   }
111 
112   @Override
113   public void setResponseMaxAge(Duration duration) {
114     long seconds = duration.getSeconds();
115     int intSeconds = seconds > Integer.MAX_VALUE ? Integer.MAX_VALUE : (int)seconds;
116     if (maxAgeLimit != null) {
117       maxAgeLimit = Math.min(maxAgeLimit, intSeconds);
118     }
119     else {
120       maxAgeLimit = intSeconds;
121     }
122   }
123 
124   /**
125    * @return the min max-age value of all responses that have been retrieved, or 365 days if no responses have been
126    *         fetched, or none of them had a max-age header
127    */
128   @Override
129   public Integer getResponseMaxAge() {
130 
131     if (maxAgeLimit == null && inputMaxAgeSeconds.isEmpty()) {
132       return null;
133     }
134 
135     int upperLimit = ObjectUtils.defaultIfNull(maxAgeLimit, (int)TimeUnit.DAYS.toSeconds(365));
136 
137     int inputMaxAge = inputMaxAgeSeconds.stream()
138         // find the max-age values of all requested resources
139         .mapToInt(triple -> Math.round(triple.getTime()))
140         // get the minimum max age time
141         .min()
142         // or fall back to the upper limit if no resources were retrieved
143         .orElse(upperLimit);
144 
145     return Math.min(inputMaxAge, upperLimit);
146   }
147 
148   List<TimeMeasurement> getSortedInputMaxAgeSeconds() {
149     return TimeMeasurement.LONGEST_TIME_FIRST.sortedCopy(inputMaxAgeSeconds);
150   }
151 
152   List<TimeMeasurement> getSortedInputResponseTimes() {
153     return TimeMeasurement.LONGEST_TIME_FIRST.sortedCopy(inputResponseTimes);
154   }
155 
156   List<TimeMeasurement> getGroupedAndSortedInvocationTimes(Class category, boolean useMax) {
157 
158     List<TimeMeasurement> invocationTimes = methodInvocationTimes.get(category.getSimpleName());
159 
160     List<TimeMeasurement> groupedInvocationTimes = new ArrayList<>();
161     invocationTimes.stream()
162         .collect(Collectors.groupingBy(TimeMeasurement::getText))
163         .forEach((text, measurements) -> {
164           DoubleStream individualTimes = measurements.stream().mapToDouble(TimeMeasurement::getTime);
165 
166           double totalTime = useMax ? individualTimes.max().orElse(0.f) : individualTimes.sum();
167           long invocations = measurements.stream().count();
168 
169           String prefix = "";
170           if (measurements.size() > 1) {
171             prefix = useMax ? "max of " : "sum of ";
172           }
173           groupedInvocationTimes.add(new TimeMeasurement(prefix + invocations + "x " + text, (float)totalTime, measurements.get(0).getUnit()));
174         });
175 
176     groupedInvocationTimes.sort(TimeMeasurement.LONGEST_TIME_FIRST);
177 
178     return groupedInvocationTimes;
179   }
180 
181   float getOverallResponseTimeMillis() {
182     return overalResponseTimeStopwatch.elapsed(TimeUnit.MICROSECONDS) / 1000.f;
183   }
184 
185   float getSumOfResponseTimeMillis() {
186     return (float)inputResponseTimes.stream()
187         .mapToDouble(m -> m.getTime())
188         .sum();
189   }
190 
191   float getSumOfInvocationMillis(Class category) {
192     return (float)methodInvocationTimes.get(category.getSimpleName()).stream()
193         .mapToDouble(m -> m.getTime())
194         .sum();
195   }
196 
197   List<Link> getSourceLinks() {
198     return ImmutableList.copyOf(sourceLinks);
199   }

201   /**
202    * @param resourceImpl the resource implementation that was used to generate the resource
203    * @return a new {@link HalResource} instance with detailed information about timing and caching for all upstream
204    *         resources accessed while handling the current request
205    */
206   @Override
207   public HalResource createMetadataResource(LinkableResource resourceImpl) {
208 
209     metadataWasRendered.set(true);
210 
211     Stopwatch stopwatch = Stopwatch.createStarted();
212 
213     HalResource metadataResource = new HalResource();
214 
215     metadataResource.getModel().put("title", "Detailed information about the performance and input data for this request");
216     if (resourceImpl != null) {
217       metadataResource.getModel().put("class", resourceImpl.getClass().getName());
218     }
219 
220     HalResource viaLinks = new HalResource().addLinks(StandardRelations.VIA, getSourceLinks());
221     addEmbedded(metadataResource, SOURCE_LINKS, viaLinks,
222         "Links to all requested upstream HAL resources (in the order the responses have been retrieved)",
223         "If you see lots of untitled resources here then free feel to add a title to the self link in that resource in the upstream service.");
224 
225     HalResource responseTimes = createTimingResource(getSortedInputResponseTimes());
226     addEmbedded(metadataResource, RESPONSE_TIMES, responseTimes,
227         "The individual response & parse times of all retrieved HAL resources",
228         "Response times > ~20ms usually indicate that the resource was not found in cache"
229             + " - a reload of this resource should then be much faster. "
230             + "If you see many individual requests here then check if the upstream "
231             + "service also provides a way to fetch this data all at once. ");
232 
233     HalResource emissionTimes = createTimingResource(getGroupedAndSortedInvocationTimes(EmissionStopwatch.class, true));
234     addEmbedded(metadataResource, EMISSION_TIMES, emissionTimes,
235         "A breakdown of emission and rendering times by resource and method",
236         "Use these times to identify performance hotspots in your server-side implementation classes");
237 
238     HalResource proxyInvocationTimes = createTimingResource(getGroupedAndSortedInvocationTimes(HalApiClient.class, false));
239     addEmbedded(metadataResource, INVOCATION_TIMES, proxyInvocationTimes,
240         "A breakdown of time spent in blocking HalApiClient proxy method calls",
241         null);
242 
243     HalResource asyncRendererResource = createTimingResource(getGroupedAndSortedInvocationTimes(AsyncHalResourceRenderer.class, false));
244     addEmbedded(metadataResource, INVOCATION_TIMES, asyncRendererResource,
245         "A breakdown of time spent in blocking method calls by AsyncHalResourceRenderer",
246         null);
247 
248     HalResource maxAgeResource = createTimingResource(getSortedInputMaxAgeSeconds());
249     addEmbedded(metadataResource, MAX_AGE, maxAgeResource,
250         "The max-age cache header values of all retrieved resources",
251         "If the max-age in this response's cache headers is lower then you expected, "
252             + "then check the resources at the very bottom of the list, because they will determine the overall max-age time.");
253 
254     // and also include the overall max-age of the response
255     metadataResource.getModel().put("maxAge", getResponseMaxAge() + " s");
256 
257     // and a summary of the important timing results
258     metadataResource.getModel().put("sumOfProxyInvocationTime", getSumOfInvocationMillis(HalApiClient.class) + "ms");
259     metadataResource.getModel().put("sumOfResourceAssemblyTime", getSumOfInvocationMillis(AsyncHalResourceRenderer.class) + "ms");
260     metadataResource.getModel().put("sumOfResponseAndParseTimes", getSumOfResponseTimeMillis() + "ms");
261     metadataResource.getModel().put("overallServerSideResponseTime", getOverallResponseTimeMillis() + "ms");
262     metadataResource.getModel().put("metadataGenerationTime", stopwatch.elapsed(TimeUnit.MILLISECONDS) + "ms");
263 
264     return metadataResource;
265   }
266 
267   private static void addEmbedded(HalResource metadataResource, String relation, HalResource toEmbed, String title, String developerHint) {
268 
269     toEmbed.getModel().put("title", title);
270 
271     if (developerHint != null) {
272       toEmbed.getModel().put("developerHint", developerHint);
273     }
274 
275     metadataResource.addEmbedded(relation, toEmbed);
276   }
277 
278   private static HalResource createTimingResource(List<TimeMeasurement> list) {
279 
280     ObjectNode model = JsonNodeFactory.instance.objectNode();
281     ArrayNode individualMetrics = model.putArray("measurements");
282 
283     list.stream()
284         .map(measurement -> measurement.getTime() + " " + TIME_UNIT_ABBRS.get(measurement.getUnit()) + " - " + measurement.getText())
285         .forEach(title -> individualMetrics.add(title));
286 
287     return new HalResource(model);
288   }

290   /**
291    * Composition of a time value with unit and description
292    */
293   public static class TimeMeasurement {
294 
295     static final Ordering<TimeMeasurement> LONGEST_TIME_FIRST = Ordering.natural().onResultOf(TimeMeasurement::getTime).reverse();
296 
297     private final String text;
298     private final Float time;
299     private final TimeUnit unit;
300 
301     TimeMeasurement(String text, Float time, TimeUnit unit) {
302       this.text = text;
303       this.time = time;
304       this.unit = unit;
305     }
306 
307     public String getText() {
308       return this.text;
309     }
310 
311     public Float getTime() {
312       return this.time;
313     }
314 
315     public TimeUnit getUnit() {
316       return this.unit;
317     }
318   }
