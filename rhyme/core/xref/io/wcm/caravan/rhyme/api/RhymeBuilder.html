20  package io.wcm.caravan.rhyme.api;
21  
22  import org.osgi.annotation.versioning.ProviderType;
23  
24  import io.wcm.caravan.rhyme.api.client.HalApiClient;
25  import io.wcm.caravan.rhyme.api.common.HalResponse;
26  import io.wcm.caravan.rhyme.api.resources.LinkableResource;
27  import io.wcm.caravan.rhyme.api.spi.ExceptionStatusAndLoggingStrategy;
28  import io.wcm.caravan.rhyme.api.spi.HalApiAnnotationSupport;
29  import io.wcm.caravan.rhyme.api.spi.HalApiReturnTypeSupport;
30  import io.wcm.caravan.rhyme.api.spi.HalResourceLoader;
31  import io.wcm.caravan.rhyme.api.spi.RhymeDocsSupport;
32  import io.wcm.caravan.rhyme.impl.RhymeBuilderImpl;
33  
34  /**
35   * A builder to configure and create a {@link Rhyme} instance to be used throughout the lifecycle of the incoming
36   * request.
37   * <p>
38   * If you are only using {@link Rhyme} as a HAL client library
39   * (but not to render HAL+JSON responses), you can also use {@link HalApiClient} directly.
40   * </p>
41   * @see Rhyme
42   */
43  @ProviderType
44  public interface RhymeBuilder {
45  
46    /**
47     * Create a {@link RhymeBuilder} that can only build {@link Rhyme} instances which do not request any resources
48     * from upstream services
49     * @return the new instance
50     * @deprecated no longer required because there is now a default {@link HalResourceLoader} implementation available,
51     *             use {@link #create()} instead
52     */
53    @Deprecated
54    static RhymeBuilder withoutResourceLoader() {
55  
56      return RhymeBuilder.create();
57    }
58  
59    /**
60     * Create a {@link RhymeBuilder} to build {@link Rhyme} instances that use a simple default HTTP client
61     * to load upstream resources. If you need to customize your HTTP request handling (e.g. authentication or caching),
62     * then use {@link #withResourceLoader(HalResourceLoader)} instead.
63     * @return the new instance
64     */
65    static RhymeBuilder create() {
66  
67      HalResourceLoader defaultLoader = HalResourceLoader.builder().build();
68  
69      return new RhymeBuilderImpl(defaultLoader);
70    }
71  
72    /**
73     * Create a {@link RhymeBuilder} to build {@link Rhyme} instances that use the given {@link HalResourceLoader}
74     * @param resourceLoader used to to load resources from upstream services
75     * @return the new instance
76     */
77    static RhymeBuilder withResourceLoader(HalResourceLoader resourceLoader) {
78  
79      return new RhymeBuilderImpl(resourceLoader);
80    }
81  
82    /**
83     * Enable generation of "curies" links (to HTML documentation generated with the rhyme-docs-maven-plugin)
84     * when rendering {@link HalResponse}s with {@link Rhyme#renderResponse(LinkableResource)}.
85     * If you call this method, you also have to ensure that you actually serve the generated HTML files using
86     * {@link RhymeDocsSupport#loadGeneratedHtml(RhymeDocsSupport, String)}
87     * @param rhymeDocsSupport the SPI instance that handles loading of the generated HTML
88     * @return this
89     */
90    RhymeBuilder withRhymeDocsSupport(RhymeDocsSupport rhymeDocsSupport);
91  
92    /**
93     * Extend the core framework to support additional return types in your annotated HAL API interfaces.
94     * You can call this method multiple times if you want to register more than one extension.
95     * @param additionalTypeSupport extension to the default type support
96     * @return this
97     */
98    RhymeBuilder withReturnTypeSupport(HalApiReturnTypeSupport additionalTypeSupport);
99  
100   /**
101    * Extend the core framework to support additional annotation types in your annotated HAL API interfaces.
102    * You can call this method multiple times if you want to register more than one extension.
103    * @param additionalTypeSupport extension to the default type support
104    * @return this
105    */
106   RhymeBuilder withAnnotationTypeSupport(HalApiAnnotationSupport additionalTypeSupport);
107 
108   /**
109    * Allow the exception handling of the core framework to support additional platform / framework specific exceptions.
110    * You can call this method multiple times if you want to register more than one extension.
111    * @param customStrategy extension to the default exception handling.
112    * @return this
113    */
114   RhymeBuilder withExceptionStrategy(ExceptionStatusAndLoggingStrategy customStrategy);
115 
116   /**
117    * Create the {@link Rhyme} instance to be used to throughout the lifecycle of an incoming request
118    * @param incomingRequestUri URI of the incoming request (can be absolute or relative, depending on the
119    *          platform/framework)
120    * @return the new instance
121    */
122   Rhyme buildForRequestTo(String incomingRequestUri);
123 
124 }
