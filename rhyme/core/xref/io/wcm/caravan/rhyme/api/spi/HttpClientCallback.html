20  package io.wcm.caravan.rhyme.api.spi;
21  
22  import java.io.InputStream;
23  import java.net.URI;
24  import java.util.Collection;
25  import java.util.Map;
26  
27  import org.osgi.annotation.versioning.ProviderType;
28  
29  import io.wcm.caravan.rhyme.api.common.HalResponse;
30  import io.wcm.caravan.rhyme.api.exceptions.HalApiClientException;
31  
32  /**
33   * Callback interface used by {@link HttpClientSupport#executeGetRequest(URI, HttpClientCallback)}
34   * to process a response that has been retrieved, and handle any errors according to the contract
35   * of the {@link HalResourceLoader#getHalResource(String)} method
36   * @see HttpClientSupport
37   * @see HalResourceLoader
38   */
39  @ProviderType
40  public interface HttpClientCallback {
41  
42    /**
43     * This should be called if your client had to modify the URL (e.g. add or replace the host
44     * name) in order to execute the request. This will ensure that the URL that was actually used will appear
45     * in any {@link HalApiClientException}s that are thrown by {@link HalResourceLoader#getHalResource(String)}
46     * @param uri from which the resource was actually requested
47     */
48    void onUrlModified(URI uri);
49  
50    /**
51     * This should be called as soon as the HTTP response headers are available. It will ensure that
52     * {@link HalResponse#getStatus()}, {@link HalResponse#getContentType()} and {@link HalResponse#getMaxAge()} will
53     * return the values from your response
54     * @param statusCode HTTP status code from the response
55     * @param headers a map with name and values of all HTTP headers in the response
56     */
57    void onHeadersAvailable(int statusCode, Map<String, ? extends Collection<String>> headers);
58  
59    /**
60     * This *must* be called after {@link #onHeadersAvailable(int, Map)} and as soon as the response body is ready to be
61     * read. It will ensure that {@link HalResponse#getBody()} will return the parsed HAL+JSON response.
62     * <p>
63     * If you don't call this method, then you *must* call {@link #onExceptionCaught(Throwable)} instead, so that
64     * the framework knows that the request has actually been completed. You *should* call this method even if the
65     * response
66     * contained an error status code, so that the framework can try to parse the response body as JSON to retrieve
67     * <a href="https://github.com/blongden/vnd.error">vnd.error</a>
68     * information. Any errors trying to read from the stream or parse the JSON will be caught and make
69     * {@link HalResourceLoader#getHalResource(String)} fail with a {@link HalApiClientException}
70     * </p>
71     * @param responseBodyStream the input stream that will be closed after the response body has been read
72     */
73    void onBodyAvailable(InputStream responseBodyStream);
74  
75    /**
76     * This *must* be called when any exception was caught that prevented you from calling
77     * {@link #onHeadersAvailable(int, Map)} and {@link #onBodyAvailable(InputStream)}. This will ensure that
78     * {@link HalResourceLoader#getHalResource(String)} fails with an {@link HalApiClientException} with all information
79     * that was already collected prior to the exception (e.g. status code).
80     * @param ex the exception that was caught by your code
81     */
82    void onExceptionCaught(Throwable ex);
83  }
