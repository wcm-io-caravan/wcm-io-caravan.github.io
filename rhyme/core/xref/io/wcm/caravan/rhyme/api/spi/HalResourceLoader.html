21  package io.wcm.caravan.rhyme.api.spi;
22  
23  import java.net.HttpURLConnection;
24  
25  import org.osgi.annotation.versioning.ConsumerType;
26  
27  import io.reactivex.rxjava3.core.Single;
28  import io.wcm.caravan.rhyme.api.Rhyme;
29  import io.wcm.caravan.rhyme.api.RhymeBuilder;
30  import io.wcm.caravan.rhyme.api.client.HalApiClient;
31  import io.wcm.caravan.rhyme.api.client.HalResourceLoaderBuilder;
32  import io.wcm.caravan.rhyme.api.common.HalResponse;
33  import io.wcm.caravan.rhyme.api.exceptions.HalApiClientException;
34  import io.wcm.caravan.rhyme.impl.client.HalResourceLoaderBuilderImpl;
35  
36  /**
37   * An interface to delegate (or mock) the actual loading (and caching) of all JSON+HAL resources
38   * loaded via {@link HalApiClient} and {@link Rhyme} to any other HTTP client
39   * client implementation. The implementation needs to be thread safe, and the same
40   * instance can be re-used throughout your application life-cycle.
41   * <p>
42   * If you don't need any configuration options (e.g. authentication), caching or asynchronous request handling,
43   * you can can simply use {@link #withDefaultHttpClient()} to create an instance that is using
44   * {@link HttpURLConnection} to execute the HTTP requests while blocking the current request.
45   * </p>
46   * <p>
47   * If you need full control over the HTTP client implementation to be used, and/or want to enable
48   * caching of HAL responses, you should call {@link #builder()} to configure and build
49   * a {@link HalResourceLoader} instance with the methods from {@link HalResourceLoaderBuilderImpl}.
50   * For caching to work properly, it is important that you'll then store and re-use the same
51   * {@link HalResourceLoader} instance when creating {@link Rhyme} or {@link HalApiClient} objects
52   * </p>
53   * <p>
54   * You can also implement this interface yourself, but this will require you to
55   * also implement the JSON parsing and exception handling according to the expectations of the
56   * {@link HalApiClient} implementation. A simpler way is to implement the callback-style
57   * {@link HttpClientSupport} interface, and then use
58   * {@link HalResourceLoaderBuilderImpl#withCustomHttpClient(HttpClientSupport)}
59   * to adapt it. In both cases, you should extend
60   * the io.wcm.caravan.rhyme.testing.client.AbstractHalResourceLoaderTest (from the test-jar)
61   * to test your implementation against a Wiremock server, to ensure that all expectations
62   * regarding response and error handling are met.
63   * </p>
64   * @see HttpClientSupport
65   * @see HalResourceLoaderBuilderImpl
66   * @see RhymeBuilder#withResourceLoader(HalResourceLoader)
67   * @see HalApiClient#create(HalResourceLoader)
68   */
69  @FunctionalInterface
70  @ConsumerType
71  public interface HalResourceLoader {
72  
73    /**
74     * Provides the response to a GET request for the given URL. The returned {@link Single} will only
75     * emit a {@link HalResponse} if the request was successful (i.e. a 200 status with a valid JSON body was returned).
76     * If anything goes wrong during the request cycle, the {@link Single} fails with {@link HalApiClientException}
77     * which gives you access to the status code and root cause of the failure.
78     * @param uri the URI of the resource to load. This is usually a fully qualified HTTP(S) URL,
79     *          but it could be any URI (depending on how the entry point was loaded and which kind of links are
80     *          represented in the upstream resources)
81     * @return a {@link Single} that emits a {@link HalResponse} entity if the request was successful, or otherwise fails
82     *         with a {@link HalApiClientException}
83     */
84    Single<HalResponse> getHalResource(String uri);
85  
86  
87    static HalResourceLoaderBuilder builder() {
88  
89      return new HalResourceLoaderBuilderImpl();
90    }
91  
92    /**
93     * Create a {@link HalResourceLoader} that uses a {@link HttpURLConnection} with default configuration to
94     * load the upstream resources.
95     * @return a {@link HalResourceLoader} that can handle fully qualified HTTP or HTTPS URIs,
96     *         blocks the current thread while loading the resource,
97     *         and does not implement any caching
98     */
99    static HalResourceLoader withDefaultHttpClient() {
100 
101     // if no further methods are called, the builder will create a loader that uses HttpURLConnection (and no caching)
102     return builder().build();
103   }
104 
105 }
