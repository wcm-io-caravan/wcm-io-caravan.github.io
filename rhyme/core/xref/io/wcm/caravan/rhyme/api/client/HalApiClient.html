21  package io.wcm.caravan.rhyme.api.client;
22  
23  import org.osgi.annotation.versioning.ProviderType;
24  
25  import io.wcm.caravan.rhyme.api.annotations.Related;
26  import io.wcm.caravan.rhyme.api.annotations.ResourceState;
27  import io.wcm.caravan.rhyme.api.common.RequestMetricsCollector;
28  import io.wcm.caravan.rhyme.api.spi.HalApiAnnotationSupport;
29  import io.wcm.caravan.rhyme.api.spi.HalApiReturnTypeSupport;
30  import io.wcm.caravan.rhyme.api.spi.JsonResourceLoader;
31  import io.wcm.caravan.rhyme.impl.client.HalApiClientImpl;
32  import io.wcm.caravan.rhyme.impl.reflection.DefaultHalApiTypeSupport;
33  
34  /**
35   * A generic HAL API client that will provide a dynamic proxy implementation of a given HAL API interface
36   * which loads data via HTTP as required when the interface methods annotated with {@link Related},
37   * {@link ResourceState} etc are called.
38   */
39  @ProviderType
40  public interface HalApiClient {
41  
42    /**
43     * @param uri the absolute path (and query parameters) of the entry point
44     * @param halApiInterface the HAL API interface class of a service's entry point resource
45     * @return a proxy implementation of the specified entry point interface
46     * @param <T> the HAL API interface type
47     */
48    <T> T getEntryPoint(String uri, Class<T> halApiInterface);
49  
50    /**
51     * @param jsonLoader implements the actual loading (and caching) of JSON/HAL resources via any HTTP client library
52     * @param metrics an instance of {@link RequestMetricsCollector} to collect performance relevant data for the current
53     *          incoming request
54     * @return an instance of {@link HalApiClient} that should be re-used for all upstream requests required by the
55     *         current incoming request
56     */
57    static HalApiClient create(JsonResourceLoader jsonLoader, RequestMetricsCollector metrics) {
58  
59      return new HalApiClientImpl(jsonLoader, metrics, new DefaultHalApiTypeSupport());
60    }
61  
62    /**
63     * @param jsonLoader implements the actual loading (and caching) of JSON/HAL resources via any HTTP client library
64     * @param metrics an instance of {@link RequestMetricsCollector} to collect performance relevant data for the current
65     *          incoming request
66     * @param annotationSupport an (optional) strategy to identify HAL API interfaces and methods that use different
67     *          annotations
68     * @param returnTypeSupport an (optional) strategy to support additional return types in your HAL API interface
69     *          methods
70     * @return an instance of {@link HalApiClient} that should be re-used for all upstream requests required by the
71     *         current incoming request
72     */
73    static HalApiClient create(JsonResourceLoader jsonLoader, RequestMetricsCollector metrics,
74        HalApiAnnotationSupport annotationSupport, HalApiReturnTypeSupport returnTypeSupport) {
75  
76      return new HalApiClientImpl(jsonLoader, metrics, DefaultHalApiTypeSupport.extendWith(annotationSupport, returnTypeSupport));
77    }
78  }
