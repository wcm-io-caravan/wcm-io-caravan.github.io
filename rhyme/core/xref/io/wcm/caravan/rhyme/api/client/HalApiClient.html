29  import io.wcm.caravan.rhyme.api.annotations.ResourceState;
30  import io.wcm.caravan.rhyme.api.common.RequestMetricsCollector;

37  /**
38   * A type-safe HAL client that will create a dynamic proxy implementation for a given URI
39   * and {@link HalApiInterface}.
40   * <p>
41   * Whenever you call a method on the proxy returned by
42   * {@link #getRemoteResource(String, Class)} that is annotated with a relevant annotation (e.g {@link Related}
43   * or {@link ResourceState}), an HTTP request to the upstream resource will be triggered,
44   * and additional client proxies for the related resources will be created based on the links in the response.
45   * This allows you to navigate through the resources simply by calling the methods defined in your HAL API
46   * interfaces. The actual loading of the upstream resources is delegated to a {@link HalResourceLoader} instance.
47   * </p>
48   * <p>
49   * The {@link HalApiClient} instance will be caching the proxy objects (based on the URL of the corresponding
50   * resources), and the return value of any method called on the proxy objects.
51   * This allows you to call the same methods multiple times without the overhead of repeated JSON parsing or
52   * duplicate HTTP requests. On the other hand you shouldn't use the same {@link HalApiClient} instance
53   * for too long (e.g. for multiple incoming requests to your server), since no invalidation of those caches
54   * will take place as long as your are using the same {@link HalApiClient} instance.
55   * </p>
56   * <p>
57   * A more persistent caching of the HAL responses, that does this needs to be implemented as a {@link HalResourceLoader}
58   * instead.
59   * </p>
60   * <p>
61   * A {@link RequestMetricsCollector} instance can be used to track all upstream resources that have been retrieved,
62   * and collect performance metrics about the interaction with the proxy objects.
63   * This is only relevant for clients created by the {@link Rhyme} instance while handling an incoming
64   * request. In that case you you shouldn't need to interact with
65   * {@link HalApiClient} directly, but use {@link Rhyme#getRemoteResource(String, Class)} instead. This ensures
66   * that the same {@link RequestMetricsCollector} instance is used throughout your incoming request.
67   * </p>
68   * <p>
69   * If you only want to consume HAL APIs and not use Rhyme to render your responses, you don't need
70   * to worry about the {@link RequestMetricsCollector}, and
71   * the easiest way to start is simply calling {@link HalApiClient#create()}.
72   * </p>
73   * @see HalApiInterface
74   * @see Related
75   * @see ResourceState
76   * @see ResourceRepresentation
77   * @see HalResourceLoader
78   */
79  @ProviderType
80  public interface HalApiClient {
81  
82    /**
83     * @param uri the absolute URI of the resource to load (usually the entry point of the HAL API)
84     * @param halApiInterface the HAL API interface class of a service's entry point resource
85     * @return a proxy implementation of the specified entry point interface
86     * @param <T> the HAL API interface type
87     */
88    <T> T getRemoteResource(String uri, Class<T> halApiInterface);
89  
90    /**
91     * Create a stand-alone {@link HalApiClient} (i.e. not to be used in the lifecycle of a {@link Rhyme} instance)
92     * that is using a default, blocking HTTP client.
93     * If you need more control over how your HTTP requests are executed and cached, use
94     * {@link #create(HalResourceLoader)} instead.
95     * @return an instance of {@link HalApiClient} that uses a default HTTP client implementation
96     */
97    static HalApiClient create() {
98  
99      return HalApiClient.create(HalResourceLoader.withDefaultHttpClient());
100   }
101 
102   /**
103    * Create a stand-alone {@link HalApiClient} (i.e. not to be used in the lifecycle of a {@link Rhyme} instance)
104    * that executes the HTTP request with the given {@link HalResourceLoader}. That loader can be configured and build
105    * using a {@link HalResourceLoaderBuilder}.
106    * @param resourceLoader implements the actual loading (and caching) of JSON/HAL resources via any HTTP client library
107    * @return an instance of {@link HalApiClient} that should be re-used for all upstream requests required by the
108    *         current incoming request
109    * @see HalResourceLoader
110    * @see HalResourceLoaderBuilder
111    */
112   static HalApiClient create(HalResourceLoader resourceLoader) {
113 
114     return new HalApiClientImpl(resourceLoader, RequestMetricsCollector.create(), new DefaultHalApiTypeSupport());
115   }
116 
117   /**
118    * @param resourceLoader implements the actual loading (and caching) of JSON/HAL resources via any HTTP client library
119    * @param metrics an instance of {@link RequestMetricsCollector} to collect performance relevant data for the current
120    *          incoming request
121    * @return an instance of {@link HalApiClient} that should be re-used for all upstream requests required by the
122    *         current incoming request
123    * @deprecated to create a stand-alone instance use {@link #create(HalResourceLoader)} instead
124    */
125   @Deprecated
126   static HalApiClient create(HalResourceLoader resourceLoader, RequestMetricsCollector metrics) {
127 
128     return new HalApiClientImpl(resourceLoader, metrics, new DefaultHalApiTypeSupport());
129   }
130 
131   /**
132    * An advanced overload {@link #create(HalResourceLoader, RequestMetricsCollector)} that allows
133    * to the client to be used with interfaces using non-standard annotations or return types.
134    * @param resourceLoader implements the actual loading (and caching) of JSON/HAL resources via any HTTP client library
135    * @param metrics an instance of {@link RequestMetricsCollector} to collect performance relevant data for the current
136    *          incoming request
137    * @param annotationSupport an (optional) strategy to identify HAL API interfaces and methods that use different
138    *          annotations
139    * @param returnTypeSupport an (optional) strategy to support additional return types in your HAL API interface
140    *          methods
141    * @return an instance of {@link HalApiClient} that should be re-used for all upstream requests required by the
142    *         current incoming request
143    */
144   static HalApiClient create(HalResourceLoader resourceLoader, RequestMetricsCollector metrics,
145       HalApiAnnotationSupport annotationSupport, HalApiReturnTypeSupport returnTypeSupport) {
146 
147     return new HalApiClientImpl(resourceLoader, metrics, DefaultHalApiTypeSupport.extendWith(annotationSupport, returnTypeSupport));
148   }
149 }
