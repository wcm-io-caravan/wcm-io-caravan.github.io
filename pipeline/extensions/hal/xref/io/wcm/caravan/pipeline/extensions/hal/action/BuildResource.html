20  package io.wcm.caravan.pipeline.extensions.hal.action;
21  
22  import org.osgi.annotation.versioning.ConsumerType;
23  
24  import io.wcm.caravan.hal.resource.HalResource;
25  import io.wcm.caravan.hal.resource.util.HalBuilder;
26  import io.wcm.caravan.pipeline.JsonPipelineAction;
27  import io.wcm.caravan.pipeline.JsonPipelineContext;
28  import io.wcm.caravan.pipeline.JsonPipelineOutput;
29  import io.wcm.caravan.pipeline.JsonPipelineOutputException;
30  import rx.Observable;
31  
32  /**
33   * A pipeline action you can use if you want to build a new {@link HalResource} based on another {@link HalResource}
34   * from the previous step's output.
35   * @deprecated use {@link CreateResource} instead
36   */
37  @Deprecated
38  //CHECKSTYLE:OFF
39  @ConsumerType
40  public abstract class BuildResource implements JsonPipelineAction {
41    //CHECKSTYLE:ON
42  
43    private final String selfHref;
44  
45    /**
46     * @param selfHref the path and query parameters to use in the output's self link
47     */
48    public BuildResource(String selfHref) {
49      this.selfHref = selfHref;
50    }
51  
52    @Override
53    public String getId() {
54      return "BUILD-RESOURCE(" + selfHref + ")";
55    }
56  
57    @Override
58    public final Observable<JsonPipelineOutput> execute(JsonPipelineOutput previousStepOutput, JsonPipelineContext pipelineContext) {
59  
60      if (!previousStepOutput.getPayload().isObject()) {
61        throw new JsonPipelineOutputException(BuildResource.class.getName()
62            + " expects the output of the previous step to be a JSON *object* output, but got "
63            + previousStepOutput.getPayload().toString());
64      }
65  
66      HalResource input = new HalResource((previousStepOutput.getPayload()));
67  
68      HalBuilder outputBuilder = new HalBuilder(selfHref);
69  
70      HalResource output = build(input, outputBuilder);
71  
72      return Observable.just(previousStepOutput.withPayload(output.getModel()));
73    }
74  
75    /**
76     * @param input the original {@link HalResource} from the previous step's output
77     * @param outputBuilder a builder that already contains a self-link, but no state
78     * @return the new HalResource that will be returned as the result of the pipeline action
79     */
80    public abstract HalResource build(HalResource input, HalBuilder outputBuilder);
81  
82  }
