54  /**
55   * Factory for HAL specific {@link JsonPipelineAction}s.
56   */
57  @ProviderType
58  public final class HalClient {
59  
60    private ServiceIdExtractor serviceIdExtractor;
61    private final CacheStrategy cacheStrategy;
62    private final Map<String, String> contextProperties;
63  
64    private final CaravanHttpRequest entryPointRequest;
65  
66    private List<JsonPipelineExceptionHandler> exceptionHandlers = Lists.newArrayList();
67    private Logger logger;
68  
69    /**
70     * @param serviceId Service ID
71     * @param cacheStrategy default cache strategy to use for all actions that fetch additional resources
72     */
73    public HalClient(String serviceId, CacheStrategy cacheStrategy) {
74      this(serviceId, cacheStrategy, ImmutableMap.of());
75    }
76  
77    /**
78     * @param serviceId Service ID
79     * @param cacheStrategy default cache strategy to use for all actions that fetch additional resources
80     * @param contextProperties a Map of properties to pass on to
81     *          {@link JsonPipelineFactory#create(CaravanHttpRequest, Map)}
82     */
83    public HalClient(String serviceId, CacheStrategy cacheStrategy, Map<String, String> contextProperties) {
84      this.serviceIdExtractor = (path) -> serviceId;
85      this.cacheStrategy = cacheStrategy;
86      this.contextProperties = contextProperties;
87      this.entryPointRequest = new CaravanHttpRequestBuilder(serviceId).append("/").build();
88    }
89  
90    /**
91     * @param entryPointRequest the request to be executed to fetch the HAL entry point
92     * @param cacheStrategy default cache strategy to use for all actions that fetch additional resources
93     * @param contextProperties a Map of properties to pass on to
94     *          {@link JsonPipelineFactory#create(CaravanHttpRequest, Map)}
95     */
96    public HalClient(CaravanHttpRequest entryPointRequest, CacheStrategy cacheStrategy, Map<String, String> contextProperties) {
97      this.serviceIdExtractor = (path) -> entryPointRequest.getServiceId();
98      this.cacheStrategy = cacheStrategy;
99      this.contextProperties = contextProperties;
100     this.entryPointRequest = entryPointRequest;
101   }
102 
103   /**
104    * @param entryPointRequest the request to be executed to fetch the HAL entry point
105    * @param serviceIdExtractor Service ID extractor
106    * @param cacheStrategy default cache strategy to use for all actions that fetch additional resources
107    * @param contextProperties a Map of properties to pass on to
108    *          {@link JsonPipelineFactory#create(CaravanHttpRequest, Map)}
109    */
110   public HalClient(CaravanHttpRequest entryPointRequest, ServiceIdExtractor serviceIdExtractor, CacheStrategy cacheStrategy,
111       Map<String, String> contextProperties) {
112     this.serviceIdExtractor = (path) -> entryPointRequest.getServiceId();
113     this.cacheStrategy = cacheStrategy;
114     this.contextProperties = contextProperties;
115     this.entryPointRequest = entryPointRequest;
116   }

148   /**
149    * Creates a {@link JsonPipeline} that will fetch the entry point of the HAL service
150    * @param factory the factory to use to create the pipeline
151    * @return the pipeline
152    */
153   public JsonPipeline getEntryPoint(JsonPipelineFactory factory) {
154     return create(factory, entryPointRequest);
155   }

157   /**

159    * @deprecated use {@link #getEntryPoint(JsonPipelineFactory)} instead
160    * @param factory Pipeline factory
161    * @return JSON pipeline
162    */
163   @Deprecated
164   public JsonPipeline createEntryPoint(JsonPipelineFactory factory) {
165     return create(factory, "/");
166   }

168   /**
169    * Creates a JSON pipeline for a service and URL.
170    * @deprecated specify the entry point in the constructor and use {@link #getEntryPoint(JsonPipelineFactory)} instead
171    * @param factory Pipeline factory
172    * @param url URL to start
173    * @return JSON pipeline
174    */
175   @Deprecated
176   public JsonPipeline create(JsonPipelineFactory factory, String url) {
177     return create(factory, new CaravanHttpRequestBuilder(serviceIdExtractor.getServiceId(url)).append(url).build());
178   }
179 
180   /**
181    * Creates a JSON pipeline for a HTTP request.
182    * @deprecated specify the entry point in the constructor and use {@link #getEntryPoint(JsonPipelineFactory)} instead
183    * @param factory Pipeline factory
184    * @param request Pre-configured HTTP request
185    * @return JSON pipeline
186    */
187   @Deprecated
188   public JsonPipeline create(JsonPipelineFactory factory, CaravanHttpRequest request) {
189     JsonPipeline entryPoint = factory.create(request, contextProperties);
190     if (cacheStrategy != null) {
191       entryPoint = entryPoint.addCachePoint(cacheStrategy);
192     }
193     return entryPoint;
194   }

325   /**
326    * Allows to create a {@link BuildResource} action by specifying the output href and a lambda
327    * @param selfHref the path of the output resource to be used for the self helf
328    * @param buildFunc the lambda that gets the previous step's output and a {@link HalBuilder} with the specified
329    *          self-link
330    * @return the action that executes the lambda build function
331    * @deprecated use createResource instead
332    **/
333   @Deprecated
334   public BuildResource buildResource(String selfHref, Func2<HalResource, HalBuilder, HalResource> buildFunc) {
335     return new BuildResource(selfHref) {
336 
337       @Override
338       public HalResource build(HalResource input, HalBuilder outputBuilder) {
339         return buildFunc.call(input, outputBuilder);
340       }
341     };
342   }
343 
344   /**
345    * Allows to create a {@link CreateResource} action by specifying the output href and a lambda.
346    * The {@link CreateResource} action will automatically generated a cache-key based on the given href, and also
347    * set the self-link of the resource returned by the lambda
348    * @param selfHref the path of the output resource to be used for the self helf
349    * @param buildFunc the lambda that gets the previous step's output and should return a new HalResource
350    * @return the action that executes the lambda build function
351    **/
352   public CreateResource createResource(String selfHref, Function<HalResource, HalResource> buildFunc) {
353     return new CreateResource(selfHref) {
354 
355       @Override
356       public HalResource createOutput(HalResource input) {
357         return buildFunc.apply(input);
358       }
359     };
360   }
361 
362   /**
363    * Allows to create a {@link ModifyResource} action by specifying the output href and a lambda
364    * @param selfHref the path of the output resource to be used for the self helf
365    * @param modifyFunc the lambda that gets a HalResource with the previous step's output and the specified self-link
366    * @return the action that executes the lambda build function
367    **/
368   public ModifyResource modifyResource(String selfHref, Action1<HalResource> modifyFunc) {
369     return new ModifyResource(selfHref) {
370 
371       @Override
372       public void modify(HalResource output) {
373         modifyFunc.call(output);
374       }
375     };
376   }
