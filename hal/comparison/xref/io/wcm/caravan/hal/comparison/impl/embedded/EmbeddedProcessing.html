20  package io.wcm.caravan.hal.comparison.impl.embedded;
21  
22  import io.wcm.caravan.hal.comparison.impl.ProcessingResult;
23  import io.wcm.caravan.hal.comparison.impl.context.HalComparisonContextImpl;
24  import io.wcm.caravan.hal.resource.HalResource;
25  
26  /**
27   * Internal interface that allows to mock/replace the processing of embedded resources for unit-tests
28   */
29  public interface EmbeddedProcessing {
30  
31    /**
32     * Processing of embedded resources happens before the state of these resources are compared, and consists of the
33     * following steps
34     * <ul>
35     * <li>Filter out resources that shouldn't be compared</li>
36     * <li>Find corresponding pairs of embedded resources (e.g. if they have been re-ordered)</li>
37     * <li>Collect differences that don't require/allow looking into the resources (e.g. resource has been removed)</li>
38     * </ul>
39     * @param context specifies in which part of the tree the comparison is currently being executed
40     * @param expected the "ground truth" resource containing the embedded resources
41     * @param actual the resource to be compared against the ground truth
42     * @return a {@link ProcessingResult} instance
43     */
44    ProcessingResult<HalResource> process(HalComparisonContextImpl context, HalResource expected, HalResource actual);
45  
46  }
